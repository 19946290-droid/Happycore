function calcularIMC() { let peso = parseFloat(document.getElementById('peso').value); let altura = parseFloat(document.getElementById('altura').value); if (!peso || !altura) { alert("Por favor ingresa tu peso y altura."); return; } let imc = peso / (altura * altura); let mensaje = ""; if (imc < 18.5) { mensaje = "Tu IMC es " + imc.toFixed(2) + " - Bajo peso"; } else if (imc < 25) { mensaje = "Tu IMC es " + imc.toFixed(2) + " - Peso normal"; } else if (imc < 30) { mensaje = "Tu IMC es " + imc.toFixed(2) + " - Sobrepeso"; } else { mensaje = "Tu IMC es " + imc.toFixed(2) + " - Obesidad"; } document.getElementById('resultado').innerText = mensaje; }
